    function twoDigit(num) {
        var ret;
        if (num < 10)

    function showClock() {
        var now = new Date();
        var year = now.getFullYear();
        var month = now.getMonth() + 1;
        var date = now.getDate();
        var week = now.getDay();
        var wd = ['日', '月', '火', '水', '木', '金', '土'];

    function twoDigit(num) {
        var ret;
        if (num < 10)
            ret = "0" + num;
        else
            ret = num;
        return ret;
    }

    function showClock() {
        var now = new Date();
        var year = now.getFullYear();
        var month = now.getMonth() + 1;
        var date = now.getDate();
        var week = now.getDay();
        var wd = ['日', '月', '火', '水', '木', '金', '土'];
        var nowHour = twoDigit(now.getHours());
        var nowMin = twoDigit(now.getMinutes());
        var nowSec = twoDigit(now.getSeconds());
        var msgDate = year + '年' + month + '月' + date+ '日'+ '(' + wd[week] + ')';
        var msgTime = "" + nowHour + ":" + nowMin + ":" + nowSec;
        //document.getElementById("date").innerHTML = year + '年' + month + '月' + date+ '日';
        //target0 = document.getElementById("today");
        //target0.innerHTML = '今日は　' + msgDate;
        //target1 = document.getElementById("now");
        //target1.innerHTML = '現在時刻　' + msgTime;
        target = document.getElementById("realtime");
        target.innerHTML =msgDate + ' ' + msgTime;
    }
    setInterval('showClock()', 1000);
</script>
